Tính chu vi và diện tích hình tròn
Source
#include <stdio.h>
#define PI 3.14159
  
int main() {
  float banKinh, chuVi, dienTich;

  printf("Nhập bán kính hình tròn: ");
  scanf("%f", &banKinh);

  //kiểm tra bán kính là số dương
  if (banKinh > 0) {
      chuVi = 2 * PI * banKinh;
      dienTich = PI * banKinh * banKinh;

      printf("Chu vi hình tròn: %.2f\n", chuVi);
      printf("Diện tích hình tròn: %.2f\n", dienTich);
  } else {
      printf("Bán kính phải là số dương.\n");
  }

  return 0;
}
Input
Nhập bán kính hình tròn: 3.5
Output
Chu vi hình tròn: 21.99
Diện tích hình tròn: 38.48
